Jenkinsfile:
node {
    def app
    stage('Clone repository') {
        checkout scm
        echo 'cloned repository.'
    }
    stage('Build image') {
        app = docker.build("gavro081/jenkins-temp", ".")
        echo 'built image.'
    }
    stage('Push image') {
        docker.withRegistry('https://registry.hub.docker.com', 'dockerhub') {
            app.push("${env.NAME_OF_BRANCH}-${env.BUILD_NUMBER}")
            app.push("${env.NAME_OF_BRANCH}-latest")
        }
        echo 'pushed image.'
    }
    echo "pipeline completed for ${env.NAME_OF_BRANCH}-${env.BUILD_NUMBER}."
}
