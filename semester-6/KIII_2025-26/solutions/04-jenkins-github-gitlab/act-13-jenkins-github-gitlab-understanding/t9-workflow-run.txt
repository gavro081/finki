# ci.yml
name: CI Demo

on:
  push:
    branches: [ main ]

jobs:
  build:
    runs-on: ubuntu-latest

    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Print message
        run: echo "Hello, GitHub Actions!"

      - name: Show Node version
        run: node -v


$ git push -> triggers action

•	Jobs: build
