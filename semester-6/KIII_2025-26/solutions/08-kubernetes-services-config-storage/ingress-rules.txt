apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: multi-rule-ingress
  annotations:
    nginx.ingress.kubernetes.io/rewrite-target: /
spec:
  ingressClassName: nginx
  rules:
  - host: app-a.local
    http:
      paths:
      - path: /
        pathType: Prefix
        backend:
          service:
            name: backend-a-svc
            port:
              number: 80
  - host: app-b.local
    http:
      paths:
      - path: /
        pathType: Prefix
        backend:
          service:
            name: backend-b-svc
            port:
              number: 80
  - host: app-a.local
    http:
      paths:
      - path: /beta
        pathType: Prefix
        backend:
          service:
            name: backend-b-svc
            port:
              number: 80
