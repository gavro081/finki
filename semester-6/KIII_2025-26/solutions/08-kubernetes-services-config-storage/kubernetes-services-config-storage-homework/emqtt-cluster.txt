Filip@filip's macbook pro dev % cat emqtt-cluster.yaml
apiVersion: v1
kind: Service
metadata:
  name: emqx-headless
  labels:
    app: emqx
spec:
  clusterIP: None
  selector:
    app: emqx
  ports:
    - name: mqtt
      port: 1883
      targetPort: 1883
    - name: mqtts
      port: 8883
      targetPort: 8883
    - name: dashboard
      port: 18083
      targetPort: 18083
    - name: ekka
      port: 4370
      targetPort: 4370
---
apiVersion: v1
kind: ConfigMap
metadata:
  name: emqx-cluster-config
  labels:
    app: emqx
data:
  EMQX_NAME: "emqx"
  EMQX_CLUSTER__DISCOVERY_STRATEGY: "static"
  EMQX_CLUSTER__STATIC__SEEDS: "[emqx@emqx-0.emqx-headless.default.svc.cluster.local,emqx@emqx-1.emqx-headless.default.svc.cluster.local,emqx@emqx-2.emqx-headless.default.svc.cluster.local]"
  EMQX_NODE__COOKIE: "emqxsecretcookie"
  EMQX_DASHBOARD__DEFAULT_USERNAME: "admin"
  EMQX_DASHBOARD__DEFAULT_PASSWORD: "public"
---
apiVersion: apps/v1
kind: StatefulSet
metadata:
  name: emqx
  labels:
    app: emqx
spec:
  serviceName: emqx-headless
  replicas: 3
  selector:
    matchLabels:
      app: emqx
  template:
    metadata:
      labels:
        app: emqx
    spec:
      containers:
        - name: emqx
          image: emqx/emqx:5.3.2
          ports:
            - name: mqtt
              containerPort: 1883
            - name: mqtts
              containerPort: 8883
            - name: dashboard
              containerPort: 18083
            - name: ekka
              containerPort: 4370
          env:
            - name: EMQX_HOST
              value: "$(POD_NAME).emqx-headless.default.svc.cluster.local"
            - name: POD_NAME
              valueFrom:
                fieldRef:
                  fieldPath: metadata.name
          envFrom:
            - configMapRef:
                name: emqx-cluster-config
          volumeMounts:
            - name: emqx-data
              mountPath: /opt/emqx/data
          readinessProbe:
            tcpSocket:
              port: 1883
            initialDelaySeconds: 10
            periodSeconds: 5
  volumeClaimTemplates:
    - metadata:
        name: emqx-data
      spec:
        accessModes: ["ReadWriteOnce"]
        resources:
          requests:
            storage: 1Gi
