# Deployment + Service for version 1.0
apiVersion: apps/v1
kind: Deployment
metadata:
  name: web-app-v1
  labels:
    app: web-app
    version: "1.0"
spec:
  replicas: 3
  selector:
    matchLabels:
      app: web-app
      version: "1.0"
  template:
    metadata:
      labels:
        app: web-app
        version: "1.0"
    spec:
      containers:
        - name: web-app
          image: gavro081/my-site:1.0
          ports:
            - containerPort: 80
---
apiVersion: v1
kind: Service
metadata:
  name: web-app-v1-svc
spec:
  type: NodePort
  selector:
    app: web-app
    version: "1.0"
  ports:
    - port: 80
      targetPort: 80
      nodePort: 30081
---
