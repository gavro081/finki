# Deployment manifest
apiVersion: apps/v1
kind: Deployment
metadata:
  name: http-deployment
spec:
  replicas: 3
  selector:
    matchLabels:
      app: http-app
  template:
    metadata:
      labels:
        app: http-app
    spec:
      containers:
      - name: http-container
        image: nginx:alpine
        ports:
        - containerPort: 80
---
# Service manifest
apiVersion: v1
kind: Service
metadata:
  name: http-nodeport-svc
spec:
  type: NodePort
  selector:
    app: http-app
  ports:
  - port: 80
    targetPort: 80
    nodePort: 30080
